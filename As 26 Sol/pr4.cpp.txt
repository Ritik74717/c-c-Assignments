#include<iostream>
using namespace std;
class Counter
{ 
  private:
       static int count;
  public:
     Counter()
     {
    
     }
     void incr()
     {
      count++;
     }
     int get()
     {
      return count;
     }
      
};
int Counter::count;
int main()
{
  Counter c1;
  cout<<c1.get();
  c1.incr();
  cout<<c1.get();

}
