#include<iostream>
using namespace std;
class Cube
{
  private:
        int a;
  public:
        Cube()
        {
            a=1;
            cout<<"Constructor Initialized\n";

        }
        void set(int x)
        {
            a=x;
        }
        void showVolume()
        {
            cout<<" VOLUME = "<<a*a*a<<endl;
        }
};

int main()
{
   Cube c1;
   c1.set(5);
   c1.showVolume();
   return 0;

}
